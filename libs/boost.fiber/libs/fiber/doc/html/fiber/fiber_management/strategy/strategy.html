#include <boost/fiber/strategy.hpp>

class strategy
{
protected:
	typedef thread_specific_ptr< fiber >	active_fiber_t;

	static active_fiber_t   active_fiber;
	fiber                   master_fiber;

	void attach( fiber &);
	void detach( fiber &);

	void switch_between( fiber &, fiber &);

	void enable_interruption( fiber &);
	bool interruption_enabled( fiber const&);

	bool is_master( fiber const&);

	bool in_state_not_started( fiber const&);
	bool in_state_ready( fiber const&);
	bool in_state_running( fiber const&);
	bool in_state_wait_for_fiber( fiber const&);
	bool in_state_wait_for_object( fiber const&);
	bool in_state_terminated( fiber const&);

	void set_state_ready( fiber &);
	void set_state_running( fiber &);
	void set_state_wait_for_fiber( fiber &);
	void set_state_wait_for_object( fiber &);
	void set_state_terminated( fiber &);

public:
	strategy();

	virtual ~strategy();
	virtual void add( fiber) = 0;
	virtual void join( fiber::id const&) = 0;
	virtual void interrupt( fiber::id const&) = 0;
	virtual void reschedule( fiber::id const&) = 0;
	virtual void cancel( fiber::id const&) = 0;
	virtual void yield( fiber::id const&) = 0;
	virtual void register_object( object::id const&) = 0;
	virtual void unregister_object( object::id const&) = 0;
	virtual void wait_for_object( object::id const&) = 0;
	virtual void object_notify_one( object::id const&) = 0;
	virtual void object_notify_all( object::id const&) = 0;
	virtual fiber release( fiber::id const&) = 0;
	virtual void migrate( fiber) = 0;
	virtual void detach_all() = 0;
	virtual bool run() = 0;
	virtual bool empty() = 0;
	virtual std::size_t size() = 0;
	virtual std::size_t ready() = 0;
};

virtual void add( fiber) = 0

Precondition:

Fiber was not assigned to another scheduler.

Effects:

Add a fiber to the scheduler.

virtual void join( fiber::id const&) = 0

Precondition:

Fiber belongs to this scheduler.

Effects:

The active fiber will yield until the fiber identified by the id has finished its execution or was interrupted.

virtual void interrupt( fiber::id const&) = 0

Precondition:

Fiber belongs to this scheduler.

Effects:

The fiber identified by the id will be interrupted by its next execution.

virtual void reschedule( fiber::id const&) = 0

Precondition:

Fiber belongs to this scheduler.

Effects:

The scheduling-algorithm will be applied uppon the fiber identified by the id again.

virtual void cancel( fiber::id const&) = 0

Precondition:

Fiber belongs to this scheduler.

Effects:

The fiber with passed identifier will be stopped, e.g. removed from the run-queue.

virtual void yield( fiber::id const&) = 0

Precondition:

Fiber is the active fiber.

Effects:

Yield active fiber and switch back to master_fiber in order to schedule another fiber from the internal storage.

virtual void register_object( object::id const&) = 0

Effects:

Register the object (mutex, condition, event-variable).

virtual void unregister_object( object::id const&) = 0

Effects:

Remove the object (mutex, condition, event-variable)i from the register.

virtual void wait_for_object( object::id const&) = 0

Effects:

The active fiber waits on object until notified.

virtual void object_notify_one( object::id const&) = 0

Effects:

Notify one fiber waiting on the object. The fiber gets ready for scheduling.

virtual void object_notify_all( object::id const&) = 0

Effects:

Notify all fibers waiting on the object. The fibers get ready for scheduling.

virtual void release( fiber::id const&) = 0

Postcondition:

The fiber is not terminated or waits on fiber or object.

Effects:

Release fiber from scheduler.

virtual void migrate( fiber) = 0

Postcondition:

The fiber is not already managed by scheduler.

Effects:

Adds fiber to scheduler.

virtual void detach_all() = 0

Effects:

Detaches all managed fibers from this.

virtual bool run() = 0

Effects:

Executes a fiber from the internal storage and removes terminated fibers. The function returns true if a fiber could be executed or a terminated fiber could be removed - otherwise false.

Throws:

boost::system::system_error if a system call failed.

virtual bool empty() = 0

Effects:

Returns true if the scheduler contains fibers (maybe runnable or waiting).

Throws:

Nothing.

virtual std::size_t size() = 0

Effects:

Returns how many fibers the scheduler contains (maybe runnable or waiting).

Throws:

Nothing.

virtual std::size_t ready() = 0

Effects:

Returns how many fibers are ready to run in the scheduler.

Throws:

Nothing.
