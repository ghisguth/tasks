Boost.Fiber provides a bounded and a unbounded fifo suitable to synchonize fibers via message passing.

Note

fifos are bound to the scheduler passed to the construtor and can only be used by fibers managed by this scheduler. 

typedef boost::fibers::unbounded_channel< int > fifo_t;

void send( fifo_t fifo)
{
    for ( int i = 0; i < 5; ++i)
        fifo.put( i);
    fifo.deactivate();
}

void recv( fifo_t fifo)
{
    boost::optional< int > value;
    while ( fifo.take( value) )
    { std::cout << "received " << * value << std::endl; }
}

boost::fibers::scheduler<> sched;
fifo_t fifo( sched);
sched.make_fiber( send, fifo);
sched.make_fiber( recv, fifo);

for (;;)
{
    while ( sched.run() );
    if ( sched.empty() ) break;
}
